## <a name="example"></a><span data-ttu-id="cb526-108">Esempio</span><span class="sxs-lookup"><span data-stu-id="cb526-108">Example</span></span>  
 <span data-ttu-id="cb526-109">L'esempio seguente genera l'errore CS0229:</span><span class="sxs-lookup"><span data-stu-id="cb526-109">The following example generates CS0229:</span></span>  
  
```csharp  
// CS0229.cs  
  
interface IList  
{  
    int Count  
    {  
        get;  
        set;  
    }  
  
    void Counter();  
}  
  
interface Icounter  
{  
    double Count  
    {  
        get;  
        set;  
    }  
}  
  
interface IListCounter : IList , Icounter {}  
  
class MyClass  
{  
    void Test(IListCounter x)  
    {  
        x.Count = 1;  // CS0229  
        // Try one of the following lines instead:  
        // ((IList)x).Count = 1;  
        // or  
        // ((Icounter)x).Count = 1;  
    }  
  
    public static void Main() {}  
}  
```
