## <a name="example"></a><span data-ttu-id="da974-107">Esempio</span><span class="sxs-lookup"><span data-stu-id="da974-107">Example</span></span>

{  
   public virtual void Finalize() {}   // CS0465

{  
   ~B() {}  
}
