## <a name="example"></a><span data-ttu-id="4ea84-109">Esempio</span><span class="sxs-lookup"><span data-stu-id="4ea84-109">Example</span></span>

{  
    IEnumerator IEnumerable.GetEnumerator()  
    {  
        yield return 0;
