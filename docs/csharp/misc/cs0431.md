 <span data-ttu-id="d99bd-107">L'esempio seguente genera l'errore CS0431:</span><span class="sxs-lookup"><span data-stu-id="d99bd-107">The following sample generates CS0431:</span></span>  
  
```  
// CS0431.cs  
using A = Outer;  
  
public class Outer   
{  
   public class Inner   
   {  
      public static void Meth() {}  
   }  
}  
  
public class MyClass  
{  
   public static void Main()  
   {  
      A::Inner.Meth();   // CS0431  
      A.Inner.Meth();   // OK  
   }  
}  
```
