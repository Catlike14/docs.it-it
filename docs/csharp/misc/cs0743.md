## <a name="example"></a><span data-ttu-id="a5dc5-107">Esempio</span><span class="sxs-lookup"><span data-stu-id="a5dc5-107">Example</span></span>

using System.Linq;  
  
public class C  
{  
    public static int Main()  
    {  
        int[] array1 = { 1, 2, 3 ,4, 5, 6,};  
        int[] array2 = { 5, 6, 7, 8, 9 };

    }  
}
