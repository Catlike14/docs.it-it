 <span data-ttu-id="c5289-107">L'esempio seguente genera l'errore CS1622:</span><span class="sxs-lookup"><span data-stu-id="c5289-107">The following sample generates CS1622:</span></span>  
  
```csharp  
// CS1622.cs  
// compile with: /target:library  
using System.Collections;  
  
class C : IEnumerable  
{  
   public IEnumerator GetEnumerator()  
   {  
      return (IEnumerator) this;  // CS1622  
      yield return this;   // OK  
   }  
}  
```
