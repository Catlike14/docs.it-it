 <span data-ttu-id="d4339-104">Una funzione <xref:System.Collections.IEnumerable.GetEnumerator%2A> , usata per abilitare l'uso delle istruzioni foreach, non può restituire un puntatore o una matrice, ma un'istanza di una classe che può fungere da enumeratore.</span><span class="sxs-lookup"><span data-stu-id="d4339-104">A <xref:System.Collections.IEnumerable.GetEnumerator%2A> function, used to enable the use of foreach statements, cannot return a pointer or array; it must return an instance of a class that is able to act as an enumerator.</span></span> <span data-ttu-id="d4339-105">I requisiti da soddisfare per fungere da enumeratore includono la proprietà public Current e il metodo public MoveNext.</span><span class="sxs-lookup"><span data-stu-id="d4339-105">The proper requirements to serve as an enumerator include a public Current property and a public MoveNext method.</span></span>

```  
// CS0202.cs  
  
public class C1  
{  
   public int Current  
   {  
      get  
      {  
         return 0;  
      }  
   }  
  
   public bool MoveNext ()  
   {  
      return false;  
   }  
  
   public static implicit operator C1 (int c1)  
   {  
      return 0;  
   }  
}  
  
public class C2  
{  
   public int Current  
   {  
      get  
      {  
         return 0;  
      }  
   }  
  
   public bool MoveNext ()  
   {  
      return false;  
   }  
  
   public C1[] GetEnumerator ()  
   // try the following line instead  
   // public C1 GetEnumerator ()  
   {  
      return null;  
   }  
}  
  
public class MainClass  
{  
   public static void Main ()  
   {  
      C2 c2 = new C2();  
  
      foreach (C1 x in c2)   // CS0202  
      {  
         System.Console.WriteLine(x.Current);  
      }  
   }  
}  
```
