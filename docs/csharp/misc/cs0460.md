```  
// CS0460.cs  
// compile with: /target:library

{  
   BaseClass() { }  
}

{  
   void F1<T>() where T : BaseClass;  
   void F2<T>() where T : struct;  
   void F3<T>() where T : BaseClass;  
}

{  
   void I.F1<T>() where T : BaseClass {}   // CS0460

}  
```
