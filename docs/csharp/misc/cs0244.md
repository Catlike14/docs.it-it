 <span data-ttu-id="0b544-106">L'esempio seguente genera l'errore CS0244:</span><span class="sxs-lookup"><span data-stu-id="0b544-106">The following sample generates CS0244:</span></span>  
  
```  
// CS0244.cs  
// compile with: /unsafe  
  
class UnsafeTest  
{  
   unsafe static void SquarePtrParam (int* p)  
   {  
      bool b = p is object;   // CS0244 p is pointer  
   }  
  
   unsafe public static void Main()  
   {  
      int i = 5;  
      SquarePtrParam (&i);  
   }  
}  
```
